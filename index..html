<script>

    let currentStep = 0; const income = 1998;

    function checkStart() { const u = document.getElementById('unit').value.trim(); const n = document.getElementById('name').value.trim(); document.getElementById('startBtn').disabled = !(u && n); }

    function nextStep(s) { document.getElementById('step' + currentStep).classList.remove('active'); currentStep = s; document.getElementById('step' + s).classList.add('active'); if(s === 7) finalize(); }

    function prevStep(s) { document.getElementById('step' + currentStep).classList.remove('active'); currentStep = s; document.getElementById('step' + s).classList.add('active'); }

    function calculate() {

        const q1 = parseFloat(document.getElementById('q1').value); const t1 = q1 * 12 * 30;

        document.getElementById('v1').innerText = q1 + " 萬"; document.getElementById('y1').innerText = (q1 * 12).toFixed(1) + " 萬"; document.getElementById('t1').innerText = t1 + " 萬";

        const q2b = parseInt(document.getElementById('q2_buy').value); const q2m = parseFloat(document.getElementById('q2_maint').value);

        const t2b_3 = q2b * 3; const t2m = q2m * 12 * 30; const t2 = t2b_3 + t2m;

        document.getElementById('v2_buy').innerText = q2b + " 萬"; document.getElementById('car30_msg').innerText = "然後一台車我們抓開10年再換，30年就需要三台車，這樣一共就是 " + t2b_3 + " 萬";

        document.getElementById('v2_maint').innerText = q2m + " 萬"; document.getElementById('y2').innerText = (q2m * 12).toFixed(1) + " 萬"; document.getElementById('t2').innerText = Math.round(t2) + " 萬";

        const q3 = parseInt(document.getElementById('q3').value); document.getElementById('v3').innerText = q3 + " 萬"; document.getElementById('y3').innerText = (q3 / 30).toFixed(1) + " 萬"; document.getElementById('t3').innerText = q3 + " 萬";

        const q4 = parseInt(document.getElementById('q4').value); document.getElementById('v4').innerText = q4 + " 萬"; document.getElementById('y4').innerText = (q4 / 30).toFixed(1) + " 萬"; document.getElementById('t4').innerText = q4 + " 萬";

        const q5 = parseInt(document.getElementById('q5').value); document.getElementById('v5').innerText = q5 + " 萬"; document.getElementById('y5').innerText = (q5 / 30).toFixed(1) + " 萬"; document.getElementById('t5').innerText = q5 + " 萬";

        const q6 = parseInt(document.getElementById('q6').value); document.getElementById('v6').innerText = q6 + " 萬"; document.getElementById('y6').innerText = (q6 / 30).toFixed(1) + " 萬"; document.getElementById('t6').innerText = q6 + " 萬";

        const b1 = income - t1; const b2 = b1 - t2; const b3 = b2 - q3; const b4 = b3 - q4; const b5 = b4 - q5; const b6 = b5 - q6;

        document.getElementById('bal1').innerText = Math.round(b1); document.getElementById('bal2').innerText = Math.round(b2); document.getElementById('bal3').innerText = Math.round(b3); document.getElementById('bal4').innerText = Math.round(b4); document.getElementById('bal5').innerText = Math.round(b5); document.getElementById('bal6').innerText = Math.round(b6);

        [1,2,3,4,5,6].forEach(n => { const el = document.getElementById('bal' + n); const val = parseInt(el.innerText); el.style.color = val < 0 ? "#FF4D4F" : "#E67E22"; });

    }

    function finalize() {

        const u = document.getElementById('unit').value; const n = document.getElementById('name').value; document.getElementById('res_header').innerText = u + " / " + n;

        const q1 = parseFloat(document.getElementById('q1').value) * 12 * 30; const q2 = (parseInt(document.getElementById('q2_buy').value)*3) + (parseFloat(document.getElementById('q2_maint').value)*12*30);

        const q3 = parseInt(document.getElementById('q3').value); const q4 = parseInt(document.getElementById('q4').value); const q5 = parseInt(document.getElementById('q5').value); const q6 = parseInt(document.getElementById('q6').value);

        const fin = income - q1 - q2 - q3 - q4 - q5 - q6;

        document.getElementById('r1').innerText = "- " + Math.round(q1) + " 萬"; document.getElementById('r2').innerText = "- " + Math.round(q2) + " 萬"; document.getElementById('r3').innerText = "- " + q3 + " 萬"; document.getElementById('r4').innerText = "- " + q4 + " 萬"; document.getElementById('r5').innerText = "- " + q5 + " 萬"; document.getElementById('r6').innerText = "- " + q6 + " 萬";

        const rf = document.getElementById('rf'); rf.innerText = Math.round(fin) + " 萬"; rf.style.color = fin < 0 ? "#FF7675" : "#2ECC71";

    }

    calculate();
